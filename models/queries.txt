CREATE TABLE parcels(
	parcel_id VARCHAR(10) PRIMARY KEY,
	from_person_name TEXT,
	to_person_name TEXT,
	color TEXT,
	weight numeric,
	service_price numeric,
	dispatched_on TEXT,
	expected_delivery TEXT,
	sender_address TEXT,
	receiver_address TEXT,
	current_location_lat numeric,
	current_location_long numeric,
	sender_phone_no VARCHAR(20),
	receiver_phone_no VARCHAR(20)
)
